02 

case class Post(title : String, likes : Int)

validateWith(hbaseConf) { hbase =>

val sql = "select post.title,post.likes from table"
val query = hbase.select[Post](sql,parse(_).extract[Post])
 given(query) { result =>

     assertEquals(3, result.size)
     assertEquals(1, result.count(p => p.likes > 2500))
  }
}
